Search locations for Spring Could Config Server Native

.important
[source,yaml]
----
spring:
  profiles:
    active: native
  cloud:
    config:
      server:
        native:
          search-locations: >
            classpath:/config,
            classpath:/db,
            classpath:/mq
----

.build, run and test
[source,bash]
----
bash gradlew clean assemble composeUp
# do some..
bash gradlew composeDown
----

.db-dev.properties config file
[source,properties]
----
# $ http :8888/db-dev.properties
# skip http metadata ...
db.name: postgres
db.password: postgres
db.server.host: 127.0.0.1
db.server.port: 5432
db.username: postgres
----

.mq-dev.yaml config file
[source,yml]
----
# $ http :8888/mq-dev.yaml
# skip http metadata ...
mq:
  password: guest
  server:
    host: 127.0.0.1
    port: 5672
  username: guest
----

.also JSON is possible too (8888/mq-dev.json)
[source,json]
